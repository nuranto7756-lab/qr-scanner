// Logika Scanner & UI untuk GitHub let currentMode = 'masuk'; let html5QrcodeScanner; function startScanner() { document.getElementById('reader').style.display = 'block'; document.getElementById('btnMulaiScan').classList.add('hidden'); document.getElementById('btnStopScan').classList.remove('hidden'); html5QrcodeScanner = new Html5QrcodeScanner("reader", { fps: 10, qrbox: { width: 250, height: 250 } }, false); html5QrcodeScanner.render(onScanSuccess, onScanFailure); } function stopScanner() { if (html5QrcodeScanner) { html5QrcodeScanner.clear().then(() => { document.getElementById('reader').style.display = 'none'; document.getElementById('btnMulaiScan').classList.remove('hidden'); document.getElementById('btnStopScan').classList.add('hidden'); }); } } function onScanSuccess(decodedText) { stopScanner(); document.getElementById('inputQR').value = decodedText; fetchItemData(decodedText); } function onScanFailure(error) { /* Abaikan error scan */ } function fetchItemData(qrCode) { document.getElementById('inputNama').value = "Mencari data..."; document.getElementById('imgPreview').innerHTML = ''; google.script.run .withSuccessHandler(item => { if(item) { document.getElementById('inputNama').value = item.nama; document.getElementById('inputSaldo').value = "SALDO: " + item.saldo; document.getElementById('imgPreview').innerHTML = item.gambar ? `` : "Gambar tidak tersedia"; } else { alert("Barang tidak ditemukan!"); document.getElementById('inputNama').value = "-- TIDAK DITEMUKAN --"; document.getElementById('imgPreview').innerHTML = "-- Kosong --"; } }) .getDataByQR(qrCode); } function switchTab(target) { const tabT = document.getElementById('tabTransaksi'), tabL = document.getElementById('tabLog'); const contentT = document.getElementById('contentTransaksi'), contentL = document.getElementById('contentLog'); if (target === 'transaksi') { tabT.className = 'w-1/2 py-3 tab-active cursor-pointer'; tabL.className = 'w-1/2 py-3 tab-inactive cursor-pointer'; contentT.classList.remove('hidden'); contentL.classList.add('hidden'); } else { tabL.className = 'w-1/2 py-3 tab-active cursor-pointer'; tabT.className = 'w-1/2 py-3 tab-inactive cursor-pointer'; contentL.classList.remove('hidden'); contentT.classList.add('hidden'); loadLogs(); } } function setMode(mode) { currentMode = mode; const bM = document.getElementById('btnMasuk'), bK = document.getElementById('btnKeluar'), dC = document.getElementById('deptContainer'), bS = document.getElementById('btnSubmit'); if (mode === 'masuk') { bM.className = 'w-1/2 py-2 rounded-md bg-green-500 text-white shadow-md flex justify-center items-center gap-2'; bK.className = 'w-1/2 py-2 rounded-md bg-gray-100 text-gray-500 border flex justify-center items-center gap-2'; dC.classList.add('hidden'); bS.style.backgroundColor = "#4ade80"; } else { bM.className = 'w-1/2 py-2 rounded-md bg-gray-100 text-gray-500 border flex justify-center items-center gap-2'; bK.className = 'w-1/2 py-2 rounded-md bg-[#dc2626] text-white shadow-md flex justify-center items-center gap-2'; dC.classList.remove('hidden'); bS.style.backgroundColor = "#dc2626"; } } function submitData() { const btn = document.getElementById('btnSubmit'); const qr = document.getElementById('inputQR').value; const nama = document.getElementById('inputNama').value; const jumlah = document.getElementById('inputJumlah').value; const dept = document.getElementById('departemen').value; if (!qr || nama.includes("--")) return alert("Scan barang dulu!"); if (currentMode === 'keluar' && !dept) return alert("Pilih Departemen!"); btn.disabled = true; btn.innerHTML = ' PROSES...'; google.script.run .withSuccessHandler(res => { alert(res); location.reload(); }) .withFailureHandler(err => { alert("Gagal: " + err.message); btn.disabled = false; btn.innerHTML = 'KONFIRMASI'; }) .submitTransaction({ qr, jumlah, jenis: currentMode, dept: currentMode === 'keluar' ? dept : "-" }); } function loadLogs() { const tbody = document.getElementById('logTableBody'); google.script.run.withSuccessHandler(data => { if (!data.length) { tbody.innerHTML = 'Kosong'; return; } tbody.innerHTML = data.map(row => ` ${row[0]} ${new Date(row[1]).toLocaleTimeString('id-ID')} ${row[2]} ${row[4]} ${row[5]} ${row[7]} `).join(''); }).getRecentLogs(); }
